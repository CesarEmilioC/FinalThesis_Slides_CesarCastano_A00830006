\subsection{Evaluation Protocol}

\begin{frame}{Evaluation metrics -- categorical}
\hypertarget{sub:metrics}{}%
\footnotesize
\textbf{\color{tecdark}Paradigm:} \alert{supervised} multi-class classification. The task is \emph{ordinal} and \emph{class-imbalanced}, so we report per-class metrics plus two aggregates.\\[2pt]
{\scriptsize For $K$ classes, with TP$_c$, FP$_c$, FN$_c$ per class $c$:}
\vspace{2pt}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.49\textwidth}
\begin{block}{Per-class\vphantom{Ag}}
\begin{minipage}[t][2.9cm][s]{\linewidth}
\centering\footnotesize
\vspace*{\fill}
$\displaystyle \text{Precision}_c = \frac{\text{TP}_c}{\text{TP}_c+\text{FP}_c}$\par
\vspace*{\fill}
$\displaystyle \text{Recall}_c = \frac{\text{TP}_c}{\text{TP}_c+\text{FN}_c}$\par
\vspace*{\fill}
$\displaystyle F_{1,c} = 2\cdot\frac{\text{Precision}_c \cdot \text{Recall}_c}{\text{Precision}_c + \text{Recall}_c}$\par
\vspace*{\fill}
\end{minipage}
\end{block}
\end{column}
\hspace{0.02\textwidth}
\begin{column}{0.49\textwidth}
\begin{block}{Aggregates\vphantom{Ag}}
\begin{minipage}[t][2.9cm][s]{\linewidth}
\centering\footnotesize
\vspace*{\fill}
$\displaystyle \text{Macro-}F_1 = \frac{1}{K}\sum_{c=1}^{K} F_{1,c}$\par
\vspace*{\fill}
$\displaystyle \text{Weighted-}F_1 = \sum_{c=1}^{K}\frac{n_c}{N}\,F_{1,c}$\par
\vspace*{\fill}
$\displaystyle \text{Accuracy} = \frac{\sum_c \text{TP}_c}{N}$\par
\vspace*{\fill}
\end{minipage}
\end{block}
\end{column}
\end{columns}
\vspace{3pt}
\begin{infoBox}\scriptsize
\textbf{Why these three:} per-class F1 surfaces minority-class behaviour; \alert{macro-F1} is our primary aggregate under imbalance; the \alert{confusion matrix} shows \emph{where} errors land -- the key check on an ordinal task.
\end{infoBox}
\end{frame}

\begin{frame}{Evaluation metrics -- ordinal complement}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
\small
The 5 quality classes carry a \alert{natural ordering}:
\[
\text{Very Low}\!<\!\text{Low}\!<\!\text{Medium}\!<\!\text{High}\!<\!\text{Excellent}
\]
Accuracy / F1 treat \emph{all} misclassifications as equally wrong. \alert{Spearman's $\rho$} captures monotonic agreement between predicted and true class rankings:
\[
\rho \;=\; \frac{\sum_i (R(y_i)\!-\!\overline{R(y)})\,(R(\hat{y}_i)\!-\!\overline{R(\hat{y})})}{\sqrt{\sum_i(R(y_i)\!-\!\overline{R(y)})^2 \sum_i(R(\hat{y}_i)\!-\!\overline{R(\hat{y})})^2}}
\]
\vspace{-4pt}
\begin{noteBox}\scriptsize
A high $\rho$ with modest accuracy means the model preserves the \emph{quality gradient} even when the exact class is wrong -- the regime our headline result demonstrates.
\end{noteBox}
\end{column}
\begin{column}{0.42\textwidth}
\begin{exampleblock}{Reading $\rho$ \emph{in our project}}
\scriptsize
\begin{itemize}\itemsep1pt
    \item \alert{$\rho = +1$:} model orders shots exactly like the coach -- ranking perfect, exact class can still miss.
    \item \alert{$\rho \approx 0$:} predictions carry \emph{no} ordinal signal -- same as random.
    \item \alert{$\rho = -1$:} systematic inverse ordering -- good called bad and vice versa.
    \item Each $\rho$ comes with a $p$-value; we require \alert{$p\!<\!0.05$} to call it meaningful.
\end{itemize}
\end{exampleblock}
\end{column}
\end{columns}
\end{frame}
