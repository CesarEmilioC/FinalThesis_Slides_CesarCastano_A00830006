% --- Comparison of 3 models -------------------------------------------------
\begin{frame}{Comparison of three temporal models}
\setlength{\tabcolsep}{6pt}
\small
\begin{center}
\begin{tabular}{l c c c}
\toprule
\textbf{Metric} & \textbf{BiLSTM (Test03)} & \textbf{BiGRU (Test01)} & \textbf{TCN (Test01)} \\
\midrule
Architecture           & BiLSTM 64/32 & BiGRU 64/32 & 3$\times$ Conv1D \\
Test set               & 97           & 97          & 97 \\
\midrule
\textbf{Accuracy}      & \textbf{29.9\%} & 26.8\%   & \textbf{29.9\%} \\
\textbf{Macro F1}      & 0.23         & 0.21        & \textbf{0.24} \\
\textbf{Weighted F1}   & \textbf{0.30} & 0.26       & \textbf{0.30} \\
Spearman $\rho$ ($p$)  & 0.03 (0.76)  & 0.12 (0.26) & 0.14 (0.18) \\
\bottomrule
\end{tabular}
\end{center}
\vspace{4pt}
\begin{itemize}\small
    \item BiLSTM and TCN tied on accuracy; TCN slightly higher macro F1 (better on \emph{Excellent}).
    \item BiGRU trails by 3.1 pp -- merged gating loses temporal information.
    \item All three overfit severely: training $\sim\!100\%$ vs.\ test $\sim\!30\%$ $\Rightarrow$ data bottleneck.
\end{itemize}
\end{frame}

\midrule

\rowcolor{tecblue!10}

MSE     & SGD(0.9)        & 27.8\% & 0.23 & 0.07 (0.50) \\
\bottomrule
\end{tabular}
\end{center}
\vspace{4pt}
\textbf{\color{tecdark}Key finding:}
\begin{itemize}\scriptsize
    \item \alert{XEnt + SGD}: $+8.3$~pp accuracy, $+0.09$ F1, $\rho\!=\!0.19$ ($p\!=\!0.06$) -- strongest single BiLSTM run.
    \item Adam overfits faster on 484 clips; SGD's smoother updates win.
    \item Combining both changes (MSE + SGD) \emph{does not} stack.
\end{itemize}

\begin{center}

0.50 & 242 / 242 & 35.5\% & 0.29 & \textbf{0.22 ($<\!0.001$)} \\
\rowcolor{tecblue!10}

\vspace{4pt}
\textbf{\color{tecdark}Counter-intuitive:}
\begin{itemize}\scriptsize
    \item Smaller training fractions \alert{win}.
    \item 0.60: \textbf{36.1\%}, $\rho\!=\!0.23$, $p\!=\!0.001$ -- the \emph{only} BiLSTM run with statistically significant $\rho$.
    \item Augmentation ($\times 2$) inflates training redundancy beyond what 100 epochs can absorb.
\end{itemize}

% --- Headline finding -------------------------------------------------------
\begin{frame}{Headline finding: \alert{method is data-limited, not method-limited}}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.46\textwidth}
\centering
\includegraphics[width=\linewidth]{./DATA/images/experimentation/split_confusion_06.pdf}\\
{\scriptsize\itshape Confusion matrix at the 60/40 split -- broader coverage of minority classes.}
\end{column}
\begin{column}{0.52\textwidth}
\begin{alertblock}{The +9 pp jump}
\small Same architecture, same labels, same hyperparameters.\\
Only the train/test fraction changed.\\
\[
26.8\%\;\Longrightarrow\;\textbf{36.1\%},\quad \rho=0.23,\; p=0.001
\]
\end{alertblock}
\vspace{4pt}
\begin{itemize}\small
    \item If the architecture or pipeline were the binding constraint, performance would be \alert{flat across splits}.
    \item The sensitivity of test metrics to training-set size is the \alert{textbook signature} of a data-limited regime.
    \item Three pipeline-level fixes (no new data needed):\\
    \scriptsize (i) reduce augmentation factor,\\
    (ii) strengthen L2 / dropout,\\
    (iii) adopt XEnt + SGD.
\end{itemize}
\end{column}
\end{columns}
\end{frame}
