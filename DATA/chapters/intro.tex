\subsection{Context \& Access Gap}

\begin{frame}{Paddle tennis: a growing sport with limited feedback tools}
\hypertarget{sub:context}{}%
\begin{columns}[T,onlytextwidth]
\begin{column}{0.56\textwidth}
\begin{itemize}
    \item One of the fastest-growing racket sports in Europe and Latin America~\cite{FIP_Report}.
    \item Doubles game on an enclosed court with playable glass walls~\cite{padel-definition}.
    \item Amateurs rely on \alert{self-evaluation} or sporadic coach feedback.
\end{itemize}
\vspace{4pt}
\begin{block}{Access barriers for amateurs}
\begin{itemize}
    \item \kv{Lesson cost}{30--50~USD/session (CO/MX).}
    \item \kv{Min.\ monthly wage}{$\sim$300--320~USD~\cite{SalMin_Co,SalMin_Mx}.}
    \item \kv{Video tracking}{$<\!1$ in $6$ courts in MTY/Cali offer motion analysis~\cite{Beelup,SmashVision,PushItReplays,ReplayMe}.}
\end{itemize}
\end{block}
\end{column}

% --- Subsection: The Shot -----------------------------------------------------
\subsection{The Bandeja Shot}

\begin{frame}{Why the \emph{bandeja} shot?}
\hypertarget{sub:bandeja}{}%
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
\begin{itemize}
    \item Defensive overhead shot used to maintain net control after a lob~\cite{PaddleTennis_Shots}.
    \item Biomechanically rich: arm extension, wrist position, impact timing, footwork.
    \item Mastery is what \alert{separates intermediate from advanced} amateurs~\cite{padel-bandeja}.
    \item Ideal first target for an automated quality-assessment system.
\end{itemize}
\vspace{4pt}
\begin{exampleblock}{Scope of this thesis}
A single shot (\emph{bandeja}), a single output (5-class quality score), one reproducible end-to-end pipeline.
\end{exampleblock}
\end{column}

\subsection{Research Design}

\begin{frame}{Hypothesis \& Research Questions}
\hypertarget{sub:research}{}%
\begin{block}{Hypothesis}
\small A supervised \alert{recurrent model}, trained on pose-estimation data from \emph{bandeja} videos, can assign \alert{one of five quality classes} (derived from expert 1--10 grades) \alert{consistent with expert assessments} under controlled conditions.
\end{block}
\vspace{4pt}
\textbf{\color{tecdark}Research Questions:}
\begin{enumerate}\small
    \item How accurately does pose estimation extract key joints across skill levels?
    \item How well does the Long Short-Term Memory (LSTM) model replicate expert-assigned scores, and with what deviation?
    \item To what extent does the model generalize to unseen performances?
    \item To what extent is the learned representation specific to the \emph{bandeja}, and under what empirical conditions could the pipeline extend to other shots?
\end{enumerate}
\end{frame}

\begin{frame}{Contributions}
\begin{enumerate}\small
    \item A novel \alert{pose-based dataset, publicly released as a benchmark} for racket-sport motion analysis: 484 \emph{bandeja} clips from 6 players, \alert{expert-graded 1--10} (raw grades stored per clip; the training pipeline maps each grade to one of 5 quality classes), plus 2D joint trajectories and metadata.
    \item Comparative evaluation of three temporal models -- \alert{Bidirectional LSTM (BiLSTM)~\cite{Hochreiter1997LSTM}, Bidirectional Gated Recurrent Unit (BiGRU)~\cite{Cho2014GRU}, Temporal Convolutional Network (TCN)~\cite{Lea2017TCN}} -- under identical pipeline, dataset, and metrics.
    \item Two sensitivity analyses (loss/optimizer grid and 5-point split sweep) that quantify how the system reacts to design choices.
    \item A fully \alert{reproducible end-to-end pipeline} (video $\rightarrow$ pose $\rightarrow$ sequence $\rightarrow$ classification) released as open-source code.
\end{enumerate}
