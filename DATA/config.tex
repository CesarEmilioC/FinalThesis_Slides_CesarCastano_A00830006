% -----------------------------------------------------------------------------
%  Custom headline: centered section + optional subsection, bigger font.
%  Replaces the Montpellier section dots / left-aligned title.
% -----------------------------------------------------------------------------
\setbeamertemplate{headline}{%
  \leavevmode%
  \begin{beamercolorbox}[wd=\paperwidth,ht=3.6ex,dp=1.6ex,sep=4pt,center]{section in head/foot}%
    \ifx\insertsection\empty\else%
      {\usebeamerfont{section in head/foot}\color{tecdark}\insertsectionhead}%
      \ifx\insertsubsection\empty\else%
        \hspace*{0.6em}\textcolor{slidegray}{\textbar}\hspace*{0.6em}%
        {\usebeamerfont{subsection in head/foot}\color{slidegray}\insertsubsectionhead}%
      \fi%
    \fi%
  \end{beamercolorbox}%
}
