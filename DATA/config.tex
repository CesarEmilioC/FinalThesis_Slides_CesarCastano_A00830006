% -----------------------------------------------------------------------------
%  Custom headline: centered section + optional subsection.
%  ht/dp generously sized to avoid clipping by the Beamer canvas top edge.
% -----------------------------------------------------------------------------
\makeatletter
\setbeamertemplate{headline}{%
  \vbox{%
    \vskip 3ex% reserve space above the line of text so ascenders don't clip
    \hbox to \paperwidth{%
      \hfill%
      \ifx\insertsectionhead\@empty\else%
        {\usebeamerfont{section in head/foot}\color{tecdark}\insertsectionhead}%
        \ifx\insertsubsectionhead\@empty\else%
          \hspace*{0.7em}\textcolor{slidegray}{\textbar}\hspace*{0.7em}%
          {\usebeamerfont{subsection in head/foot}\color{slidegray}\insertsubsectionhead}%
        \fi%
      \fi%
      \hfill%
    }%
    \vskip 0.6ex%
    \hbox to \paperwidth{\hfill{\color{tecblue!35}\rule{0.92\paperwidth}{0.4pt}}\hfill}%
    \vskip 0.4ex%
  }%
}
\makeatother
